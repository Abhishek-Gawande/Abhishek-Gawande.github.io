#1
Two Sum
Easy Hash Table
+
Given an array of integers and a target, return indices of two numbers that add up to target.
Approach: Use HashMap to store complement → index mapping
public int[] twoSum(int[] nums, int target) {
    Map<Integer, Integer> map = new HashMap<>();
    for (int i = 0; i < nums.length; i++) {
        int complement = target - nums[i];
        if (map.containsKey(complement))
            return new int[]{map.get(complement), i};
        map.put(nums[i], i);
    }
    return new int[]{};
}
Time: O(n)
Space: O(n)
#2
Best Time to Buy & Sell Stock
Easy Greedy
+
Find max profit from one buy and one sell transaction.
Approach: Track min price seen so far, calculate profit at each step
public int maxProfit(int[] prices) {
    int minPrice = Integer.MAX_VALUE;
    int maxProfit = 0;
    for (int price : prices) {
        minPrice = Math.min(minPrice, price);
        maxProfit = Math.max(maxProfit, price - minPrice);
    }
    return maxProfit;
}
Time: O(n)
Space: O(1)

#4
Product of Array Except Self
Medium Prefix/Suffix
+
Return array where result[i] = product of all elements except nums[i] (no division allowed).
Approach: Left products × Right products in two passes
public int[] productExceptSelf(int[] nums) {
    int n = nums.length;
    int[] res = new int[n];
    res[0] = 1;
    for (int i = 1; i < n; i++)
        res[i] = res[i-1] * nums[i-1];
    int right = 1;
    for (int i = n-1; i >= 0; i--) {
        res[i] *= right;
        right *= nums[i];
    }
    return res;
}
Time: O(n)
Space: O(1)
#5
Maximum Subarray (Kadane's)
Medium DP
+
Find contiguous subarray with largest sum.
Approach: Kadane's algorithm - at each position decide to extend or restart
public int maxSubArray(int[] nums) {
    int maxSum = nums[0], curSum = nums[0];
    for (int i = 1; i < nums.length; i++) {
        curSum = Math.max(nums[i], curSum + nums[i]);
        maxSum = Math.max(maxSum, curSum);
    }
    return maxSum;
}
Time: O(n)
Space: O(1)

public void moveZeroes(int[] nums) {
    int insertPos = 0;
    for (int num : nums) {
        if (num != 0) {
            nums[insertPos++] = num;

}

#7
3Sum
Medium Two Pointers
+
Find all unique triplets that sum to zero.
Approach: Sort + fix one element + two pointers on remainder
public List<List<Integer>> threeSum(int[] nums) {
    Arrays.sort(nums);
    List<List<Integer>> res = new ArrayList<>();
    for (int i = 0; i < nums.length-2; i++) {
        if (i > 0 && nums[i] == nums[i-1]) continue;
        int l = i+1, r = nums.length-1;
        while (l < r) {
            int sum = nums[i] + nums[l] + nums[r];
            if (sum == 0) {
                res.add(Arrays.asList(nums[i], nums[l], nums[r]));
                while (l < r && nums[l] == nums[l+1]) l++;
                while (l < r && nums[r] == nums[r-1]) r--;
                l++; r--;
            } else if (sum < 0) l++;
            else r--;
        }
    }
    return res;
}
Time: O(n²)
Space: O(1)

public int maxArea(int[] height) {
    int l = 0, r = height.length - 1, maxArea = 0;
    while (l < r) {
        int area = Math.min(height[l], height[r]) * (r - l);
        maxArea = Math.max(maxArea, area);

        else r--;
    }

}

Merge Sorted Array
Easy Two Pointers
+
Merge nums2 into nums1 (which has extra space).
Approach: Fill from end to avoid overwriting
public void merge(int[] nums1, int m, int[] nums2, int n) {
    int i = m-1, j = n-1, k = m+n-1;
    while (j >= 0) {
        if (i >= 0 && nums1[i] > nums2[j])
            nums1[k--] = nums1[i--];
        else
            nums1[k--] = nums2[j--];
    }
}
Time: O(m+n)
Space: O(1)

public List<Integer> findDuplicates(int[] nums) {
    List<Integer> res = new ArrayList<>();
    for (int num : nums) {
        int idx = Math.abs(num) - 1;
        if (nums[idx] < 0) res.add(Math.abs(num));
        nums[idx] = -nums[idx];

}

#13
Rotate Array
Medium Reversal
+
Rotate array to right by k steps.
Approach: Reverse all → reverse first k → reverse last n-k
public void rotate(int[] nums, int k) {
    k %= nums.length;
    reverse(nums, 0, nums.length-1);
    reverse(nums, 0, k-1);
    reverse(nums, k, nums.length-1);
}
void reverse(int[] a, int l, int r) {
    while (l < r) {
        int t = a[l]; a[l++] = a[r]; a[r--] = t;
    }
}
Time: O(n)
Space: O(1)
#14
Search in Rotated Sorted Array
Medium Binary Search
+
Search target in rotated sorted array in O(log n).
Approach: Modified binary search - determine which half is sorted
public int search(int[] nums, int target) {
    int l = 0, r = nums.length - 1;
    while (l <= r) {
        int mid = l + (r - l) / 2;
        if (nums[mid] == target) return mid;
        if (nums[l] <= nums[mid]) {
            if (target >= nums[l] && target < nums[mid])
                r = mid - 1;
            else l = mid + 1;
        } else {
            if (target > nums[mid] && target <= nums[r])
                l = mid + 1;
            else r = mid - 1;
        }
    }
    return -1;
}
Time: O(log n)
Space: O(1)

#17
Sort Colors (Dutch Flag)
Medium Three Pointers
+
Sort array with only 0s, 1s, 2s in one pass.
Approach: Three pointers - low, mid, high
public void sortColors(int[] nums) {
    int low = 0, mid = 0, high = nums.length - 1;
    while (mid <= high) {
        if (nums[mid] == 0) {
            swap(nums, low++, mid++);
        } else if (nums[mid] == 2) {
            swap(nums, mid, high--);
        } else {
            mid++;
        }
    }
}
void swap(int[] a, int i, int j) {
    int t = a[i]; a[i] = a[j]; a[j] = t;
}
Time: O(n)
Space: O(1)
#18
Next Permutation
Medium Array Manipulation
+
Rearrange to next lexicographically greater permutation.
Approach: Find rightmost ascending pair → swap → reverse suffix
public void nextPermutation(int[] nums) {
    int i = nums.length - 2;
    while (i >= 0 && nums[i] >= nums[i+1]) i--;
    if (i >= 0) {
        int j = nums.length - 1;
        while (nums[j] <= nums[i]) j--;
        swap(nums, i, j);
    }
    reverse(nums, i+1, nums.length-1);
}
Time: O(n)
Space: O(1)

Subarray Sum Equals K
Medium Prefix Sum
+
Count subarrays with sum equal to k.
Approach: Prefix sum + HashMap
public int subarraySum(int[] nums, int k) {
    Map<Integer, Integer> map = new HashMap<>();
    map.put(0, 1);
    int sum = 0, count = 0;
    for (int num : nums) {
        sum += num;
        count += map.getOrDefault(sum - k, 0);
        map.merge(sum, 1, Integer::sum);
    }
    return count;
}
Time: O(n)

Intersection of Two Arrays II

Find intersection including duplicates.
public int[] intersect(int[] nums1, int[] nums2) {
    Map<Integer,Integer> map = new HashMap<>();
    for (int n : nums1) map.merge(n, 1, Integer::sum);
    List<Integer> res = new ArrayList<>();
    for (int n : nums2) {
        if (map.getOrDefault(n,0) > 0) {
            res.add(n);
            map.merge(n, -1, Integer::sum);
        }
    }
    return res.stream().mapToInt(i->i).toArray();
}

Valid Anagram
Easy Frequency Count
+
Check if two strings are anagrams.
Approach: Frequency array [26] or HashMap
public boolean isAnagram(String s, String t) {
    if (s.length() != t.length()) return false;
    int[] count = new int[26];
    for (int i = 0; i < s.length(); i++) {
        count[s.charAt(i) - 'a']++;
        count[t.charAt(i) - 'a']--;
    }
    for (int c : count)
        if (c != 0) return false;
    return true;
}
Time: O(n)
Space: O(1)
#2
Valid Palindrome
Easy Two Pointers
+
Check if string is palindrome (alphanumeric only, case-insensitive).
public boolean isPalindrome(String s) {
    int l = 0, r = s.length() - 1;
    while (l < r) {
        while (l<r && !Character.isLetterOrDigit(s.charAt(l))) l++;
        while (l<r && !Character.isLetterOrDigit(s.charAt(r))) r--;
        if (Character.toLowerCase(s.charAt(l)) != 
            Character.toLowerCase(s.charAt(r)))
            return false;
        l++; r--;
    }
    return true;
}

#5
Longest Substring Without Repeating Characters
Medium Sliding Window
+
Find length of longest substring without duplicates.
public int lengthOfLongestSubstring(String s) {
    Map<Character, Integer> map = new HashMap<>();
    int max = 0, left = 0;
    for (int right = 0; right < s.length(); right++) {
        char c = s.charAt(right);
        if (map.containsKey(c))
            left = Math.max(left, map.get(c) + 1);
        map.put(c, right);
        max = Math.max(max, right - left + 1);
    }
    return max;
}

public List<List<String>> groupAnagrams(String[] strs) {
    Map<String, List<String>> map = new HashMap<>();
    for (String s : strs) {
        char[] ca = s.toCharArray();
        Arrays.sort(ca);
        String key = new String(ca);
        map.computeIfAbsent(key, k->new ArrayList<>()).add(s);
    }
    return new ArrayList<>(map.values());
}

public String longestCommonPrefix(String[] strs) {
    if (strs.length == 0) return "";
    String prefix = strs[0];
    for (int i = 1; i < strs.length; i++) {
        while (!strs[i].startsWith(prefix))
            prefix = prefix.substring(0, prefix.length()-1);
    }
    return prefix;
}

public ListNode reverseList(ListNode head) {
    ListNode prev = null, curr = head;
    while (curr != null) {
        ListNode next = curr.next;
        curr.next = prev;
        prev = curr;
        curr = next;
    }
    return prev;
}
